                  /*
                   * A recursive in-order traversal. Takes a callback function, process, which is applied to each node.
                   */
                  printInOrder(process) {
                    let inOrder = (node) => {
                      if (node.left !== null) {
                        inOrder(node.left);
                      }

                      process.call(this, node);

                      if (node.right !== null) {
                        inOrder(node.right);
                      }
                    };

                    inOrder(this.root);
                  }

                  /*
                   * A recursive pre-order traversal.

                  printPreOrder(process) {
                    let preOrder = (node) => {
                      process.call(this, node);

                      if (node.left !== null) {
                        preOrder(node.left);
                      }

                      if (node.right !== null) {
                        preOrder(node.right);
                      }
                    }

                    preOrder(this.root);

                        postOrder(node.right);

                    postOrder(this.root);

                  isBalanced(){
